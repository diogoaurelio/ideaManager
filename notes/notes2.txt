# Adding associations to model

# Rating.rb
class Rating < ActiveRecord::Base
  belongs_to :idea
  belongs_to :user
end

# Idea.rb
class Idea < ActiveRecord::Base
  has_many :ratings
  belongs_to :user
end

# User.rb
class User < ActiveRecord::Base
  has_many :ratings
  has_many :ideas
end

# Routes
